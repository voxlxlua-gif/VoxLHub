local function equipTool(name)
    local char = player.Character or player.CharacterAdded:Wait()

    if tool then
        tool.Parent = char
        return tool
    end
end

-- TELEPORT

    local char = player.Character or player.CharacterAdded:Wait()
    local root = char:WaitForChild("HumanoidRootPart")
    root.CFrame = CFrame.new(x,y,z)
end

-- AUTO BLOCK
local function blockPlayers()
    for _,plr in pairs(Players:GetPlayers()) do
        if plr ~= player then
            pcall(function()
                StarterGui:SetCore("PromptBlockPlayer", plr.UserId)
            end)

        end
    end
end

-- ==================== AUTO GRAB SYSTEM (0.1 DETIK) ====================
local autoGrabEnabled = false
local allAnimalsCache = {}
local PromptMemoryCache = {}
local InternalStealCache = {}
local stealConnection = nil
local IsStealing = false
local StealProgress = 0

local function getHRP_()
    local char = LocalPlayer.Character
    if not char then return nil end
    return char:FindFirstChild("HumanoidRootPart") or char:FindFirstChild("UpperTorso")
end

local function isMyBase(plotName)
    local plot = workspace.Plots:FindFirstChild(plotName)
    if not plot then return false end
    local sign = plot:FindFirstChild("PlotSign")
    if sign and sign:FindFirstChild("YourBase") and sign.YourBase:IsA("BillboardGui") then
        return sign.YourBase.Enabled == true
    end
    return false
end

-- Fungsi untuk scan semua animal
local function scanAllAnimals()
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return end
    
    allAnimalsCache = {}
    
    for _, plot in ipairs(plots:GetChildren()) do
        if not plot:IsA("Model") then continue end
        if isMyBase(plot.Name) then continue end
        
        local podiums = plot:FindFirstChild("AnimalPodiums")
        if not podiums then continue end
        
        for _, podium in ipairs(podiums:GetChildren()) do
            if not podium:IsA("Model") then continue end
            
            local base = podium:FindFirstChild("Base")
            if not base then continue end
            
            local spawn = base:FindFirstChild("Spawn")
            if not spawn then continue end
            
            local animalName = "Unknown"
            for _, child in ipairs(spawn:GetChildren()) do
                if child:IsA("Model") and child.Name ~= "PromptAttachment" then
                    animalName = child.Name
                    break
                end
            end
            
            local worldPos = podium:GetPivot().Position
            
            table.insert(allAnimalsCache, {
                name = animalName,
                plot = plot.Name,
                slot = podium.Name,
                worldPosition = worldPos,
                uid = plot.Name .. "_" .. podium.Name,
            })
        end
    end
end

local function findProximityPromptForAnimal(animalData)
    if not animalData then return nil end
    
    local cachedPrompt = PromptMemoryCache[animalData.uid]
    if cachedPrompt and cachedPrompt.Parent then
        return cachedPrompt
    end
    
    local plot = workspace.Plots:FindFirstChild(animalData.plot)
    if not plot then return nil end
    
    local podiums = plot:FindFirstChild("AnimalPodiums")
    if not podiums then return nil end
    
    local podium = podiums:FindFirstChild(animalData.slot)
    if not podium then return nil end
    
    local base = podium:FindFirstChild("Base")
    if not base then return nil end
    
    local spawn = base:FindFirstChild("Spawn")
    if not spawn then return nil end
    
    local attach = spawn:FindFirstChild("PromptAttachment")
    if not attach then return nil end
    
    for _, p in ipairs(attach:GetChildren()) do
        if p:IsA("ProximityPrompt") then
            PromptMemoryCache[animalData.uid] = p
            return p
        end
    end
    
    return nil
end

local function buildStealCallbacks(prompt)
    if InternalStealCache[prompt] then return end
    
    local data = {
        holdCallbacks = {},
        triggerCallbacks = {},
        ready = true,
    }
    
    local success, connections = pcall(getconnections, prompt.PromptButtonHoldBegan)
    if success and type(connections) == "table" then
        for _, conn in ipairs(connections) do
            if type(conn.Function) == "function" then
                table.insert(data.holdCallbacks, conn.Function)
            end
        end
    end
    
    local success2, connections2 = pcall(getconnections, prompt.Triggered)
    if success2 and type(connections2) == "table" then
        for _, conn in ipairs(connections2) do
            if type(conn.Function) == "function" then
                table.insert(data.triggerCallbacks, conn.Function)
            end
        end
    end
    
    if (#data.holdCallbacks > 0) or (#data.triggerCallbacks > 0) then
        InternalStealCache[prompt] = data
    end
end

local function executeInternalStealAsync(prompt)
    local data = InternalStealCache[prompt]
    if not data or not data.ready then return false end
    
    data.ready = false
    IsStealing = true
    StealProgress = 0
    
    task.spawn(function()
        if #data.holdCallbacks > 0 then
            for _, fn in ipairs(data.holdCallbacks) do
                pcall(fn)
            end
        end
        
        local startTime = tick()
        while tick() - startTime < 0.1 do
            StealProgress = (tick() - startTime) / 0.1
            task.wait()
        end
        StealProgress = 1
        
        if #data.triggerCallbacks > 0 then
            for _, fn in ipairs(data.triggerCallbacks) do
                pcall(fn)
            end
        end
        
        task.wait(0.05)
        data.ready = true
        IsStealing = false
        StealProgress = 0
    end)
    
    return true
end

local function attemptSteal(prompt)
    if not prompt or not prompt.Parent then return false end
    buildStealCallbacks(prompt)
    if not InternalStealCache[prompt] then return false end
    return executeInternalStealAsync(prompt)
end

local function getNearestAnimal()
    local hrp = getHRP_()
    if not hrp then return nil end
    
    local nearest = nil
    local minDist = math.huge
    
    for _, animalData in ipairs(allAnimalsCache) do
        if isMyBase(animalData.plot) then continue end
        if animalData.worldPosition then
            local dist = (hrp.Position - animalData.worldPosition).Magnitude
            if dist < minDist then
                minDist = dist
                nearest = animalData
            end
        end
    end
    
    return nearest
end

local function autoStealLoop()
    if stealConnection then stealConnection:Disconnect() end
    
    stealConnection = RunService.Heartbeat:Connect(function()
        if not autoGrabEnabled then return end
        if IsStealing then return end
        
        local targetAnimal = getNearestAnimal()
        if not targetAnimal then return end
        
        local hrp = getHRP_()
        if not hrp then return end
        
        local distance = (hrp.Position - targetAnimal.worldPosition).Magnitude
        if distance > 25 then return end
        
        local prompt = PromptMemoryCache[targetAnimal.uid]
        if not prompt or not prompt.Parent then
            prompt = findProximityPromptForAnimal(targetAnimal)
        end
        
        if prompt then
            attemptSteal(prompt)
        end
    end)
end

-- ==================== ESP ====================
local espEnabled = false
local espLines = {}
local espTexts = {}

local function clearESP()
    for _, obj in pairs(espLines) do if obj then obj:Remove() end end
    for _, obj in pairs(espTexts) do if obj then obj:Remove() end end
    espLines = {}; espTexts = {}
end

local function updateESP()
    if not espEnabled then
        clearESP()
        return
    end
    local myPos = getHRP_()
    if not myPos then
        clearESP()
        return
    end
    myPos = myPos.Position
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
            local targetPos = player.Character.HumanoidRootPart.Position
            local dist = (myPos - targetPos).Magnitude
            if not espLines[player] then
                local line = Drawing.new("Line")
                line.Thickness = 1
                line.Color = theme.purple
                line.Transparency = 1
                line.Visible = true
                espLines[player] = line
            end
            local line = espLines[player]
            if line then
                local myScr, on1 = workspace.CurrentCamera:WorldToViewportPoint(myPos)
                local tarScr, on2 = workspace.CurrentCamera:WorldToViewportPoint(targetPos)
                if on1 and on2 then
                    line.From = Vector2.new(myScr.X, myScr.Y)
                    line.To = Vector2.new(tarScr.X, tarScr.Y)
                    line.Visible = true
                else
                    line.Visible = false
                end
            end
            if not espTexts[player] then
                local text = Drawing.new("Text")
                text.Color = theme.purple
                text.Size = 16
                text.Center = true
                text.Outline = true
                text.Visible = true
                espTexts[player] = text
            end
            local text = espTexts[player]
            if text then
                local tarScr, on = workspace.CurrentCamera:WorldToViewportPoint(targetPos)
                if on then
                    text.Position = Vector2.new(tarScr.X, tarScr.Y - 30)
                    text.Text = string.format("%.1f m", dist)
                    text.Visible = true
                else
                    text.Visible = false
                end
            end
        end
    end
end
RunService.RenderStepped:Connect(updateESP)

-- ==================== NIGHT MODE ====================
local nightMode = false
local nightModeThread = nil
local function setNightMode(active)
    nightMode = active
    if active then
        Lighting.Ambient = Color3.new(0,0,0)
        Lighting.Brightness = 0
        Lighting.OutdoorAmbient = Color3.new(0,0,0)
        Lighting.ClockTime = 0
        Lighting.GlobalShadows = false
        if nightModeThread then task.cancel(nightModeThread) end
        nightModeThread = task.spawn(function()
            while nightMode do
                task.wait(1)
                Lighting.Ambient = Color3.new(0,0,0)
                Lighting.Brightness = 0
            end
        end)
    else
        Lighting.Ambient = Color3.new(0.5,0.5,0.5)
        Lighting.Brightness = 1
        Lighting.OutdoorAmbient = Color3.new(0.5,0.5,0.5)
        if nightModeThread then task.cancel(nightModeThread) end
    end
end

-- Fungsi untuk mengecek apakah suatu objek berada di dalam plot yang dikecualikan
local function isInExcludedPlot(obj)
    local ancestor = obj
    while ancestor do
        if ancestor.Name and EXCLUDED_PLOTS[ancestor.Name] then
            return true
        end
        ancestor = ancestor.Parent
    end
    return false
end

-- Fungsi untuk mengubah objek jadi hitbox (dengan pengecekan)
local function makeHitbox(obj)
    if isInExcludedPlot(obj) then
        return false  -- Skip jika di plot yang dikecualikan
    end
    
    pcall(function()
        -- Set transparency (hitbox effect)
        obj.Transparency = TRANSPARENCY
        
        -- Set warna
        obj.Color = HITBOX_COLOR
        if obj:IsA("BasePart") then
            obj.BrickColor = BrickColor.new("Bright violet")
        end
        
        -- Set material
        if USE_NEON then
            obj.Material = Enum.Material.Neon
        else
            obj.Material = Enum.Material.SmoothPlastic
        end
        
        -- Set reflectance
        obj.Reflectance = 0.1
        
        -- Matikan shadow biar ringan
        obj.CastShadow = false
    end)
    return true
end

local totalItems = 0
local changedParts = 0
local changedItems = 0
local skippedParts = 0

for _, obj in ipairs(Workspace:GetDescendants()) do

        if not isInExcludedPlot(obj) then
            totalParts = totalParts + 1

            skippedParts = skippedParts + 1
        end
    elseif CHANGE_ITEMS and (obj:IsA("Tool") or obj:IsA("Accessory") or obj:IsA("Part") or obj:IsA("MeshPart")) then
        if not isInExcludedPlot(obj) then
            totalItems = totalItems + 1
        end
    end
end

-- ==================== AUTO KICK SYSTEM ====================
local KEYWORD = "you stole"
local KICK_MESSAGE = "VoxLHubEZ"
local connections = {}
local autoKickEnabled = false

local function hasKeyword(text)
    if typeof(text) ~= "string" then return false end
    return string.find(string.lower(text), KEYWORD) ~= nil
end

local function kick()
    pcall(function()
        player:Kick(KICK_MESSAGE)
    end)
end

local function watchObject(obj)
    if not (obj:IsA("TextLabel") or obj:IsA("TextButton") or obj:IsA("TextBox")) then
        return
    end
    if hasKeyword(obj.Text) then
        kick()
        return
    end
    local conn = obj:GetPropertyChangedSignal("Text"):Connect(function()
        if hasKeyword(obj.Text) then
            kick()
        end
    end)
    table.insert(connections, conn)
end

local function scan(parent)
    for _, obj in ipairs(parent:GetDescendants()) do
        watchObject(obj)
    end
end

local function watchGui(gui)
    scan(gui)
    local conn = gui.DescendantAdded:Connect(function(desc)
        watchObject(desc)
    end)
    table.insert(connections, conn)
end

local function startAutoKick()
    if autoKickEnabled then return end
    autoKickEnabled = true
    
    -- Bersihkan koneksi lama
    for _, conn in ipairs(connections) do
        pcall(function() conn:Disconnect() end)
    end
    connections = {}
    
    -- Mulai monitor
    for _, gui in ipairs(PlayerGui:GetChildren()) do
        watchGui(gui)
    end
    
    table.insert(connections,
        PlayerGui.ChildAdded:Connect(function(gui)
            watchGui(gui)
        end)
    )
end

local function stopAutoKick()
    autoKickEnabled = false
    for _, conn in ipairs(connections) do
        pcall(function() conn:Disconnect() end)
    end
    connections = {}
end
